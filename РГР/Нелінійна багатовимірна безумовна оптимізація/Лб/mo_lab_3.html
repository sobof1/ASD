import numpy as np
from scipy.optimize import minimize, root
import matplotlib.pyplot as plt

#     
def f2(x):
    return 5*(x[0] - 6)**2 + 3*(x[1] - 2)**3 - 6*x[0]*x[1]

#  
def grad_f2(x):
    return np.array([
        10*(x[0] - 6) - 6*x[1],  # f/x
        9*(x[1] - 2)**2 - 6*x[0]   # f/x
    ])

#  
def hess_f2(x):
    return np.array([
        [10, -6],
        [-6, 18*(x[1] - 2)]
    ])

print("\n" + "=" * 70)
print("  -   ")
print("=" * 70)
print(f": f(X) = 5(x - 6) + 3(x - 2) - 6xx")
print("-" * 70)

#  1: '  f(x) = 0   root()
print("\n1. '  f(x) = 0 ( root):")

#    root()
def equations(vars):
    x1, x2 = vars
    eq1 = 10*(x1 - 6) - 6*x2
    eq2 = 9*(x2 - 2)**2 - 6*x1
    return [eq1, eq2]

#      
initial_guesses = [[5, 0], [10, 5], [8, 4], [6, 1]]

stationary_points = []
for i, guess in enumerate(initial_guesses):
    solution = root(equations, guess, method='hybr')
    if solution.success:
        point = solution.x
        # ,       
        is_new = True
        for sp in stationary_points:
            if np.linalg.norm(point - sp) < 1e-6:
                is_new = False
                break
        
        if is_new:
            stationary_points.append(point)
            print(f"\n     {guess}:")
            print(f"     : x = [{point[0]:.6f}, {point[1]:.6f}]")
            print(f"   f(x) = [{grad_f2(point)[0]:.2e}, {grad_f2(point)[1]:.2e}]")
            print(f"   ||f(x)|| = {np.linalg.norm(grad_f2(point)):.2e}")

#  2:  ||f(x)||
print("\n2.  ||f(x)||:")

def grad_norm_squared(x):
    grad = grad_f2(x)
    return np.sum(grad**2)
